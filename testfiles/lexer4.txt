# Program to display the Fibonacci sequence up to n-th term where n is provided by the user

println("Fibonacci Sequence Generator");

# first two terms
n1 = 0;
n2 = 1;
nth = readint("Enter the amount of terms you would like to generate: ");

count = 2;
print(n1);
print(", ");
print(n2);
print(", ");

while count < nth do
    total = n1 + n2;
    print(total);
    print(", ");
    # update values
    n1 = n2;
    n2 = total;
    count = count + 1;
end
